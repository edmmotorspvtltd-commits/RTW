	 function decimalRound(num,scale){
			if($.isNumeric(num)){
				if(!("" + num).includes("e")) {
				    return +(Math.round(num + "e+" + scale)  + "e-" + scale);
				  } else {
				    var arr = ("" + num).split("e");
				    var sig = ""
				    if(+arr[1] + scale > 0) {
				      sig = "+";
				    }
				    return +(Math.round(+arr[0] + "e" + sig + (+arr[1] + scale)) + "e-" + scale);
				  }
		 	}else{
		 		return parseFloat(0).toFixed(scale);
		 	}
		} 
	 function decimalRoundoff(num)
	 {
		
		 if(num!=null)
		 {
			 if(Math.round((parseFloat(num)%0.2)*100) %2 !=0)
		    	{
				 return num=num+0.01;
		    	}
			 else{
				 return num;
			 }
		 }
		 else {
			 return 0;
		 }
	 }
	 
	 (function waitForDatePicker() {
		 
		    var attempts = 0;
		    var maxAttempts = 50;

		    var interval = setInterval(function () {
		        attempts++;

		        var entryFields = $(".entry-date");
		        var editFields = $(".edit-date");

		        if (entryFields.length === 0 && editFields.length === 0) {
		            if (attempts >= maxAttempts) clearInterval(interval);
		            return;
		        }

		        var allReady = true;
		        $(".entry-date, .edit-date").each(function () {
		            if (!$(this).data("datepicker")) {
		                allReady = false;
		            }
		        });

		        if (!allReady) {
		            if (attempts >= maxAttempts) clearInterval(interval);
		            return;
		        }

		     // ENTRY MODE: allow last N days up to today
		        $(".entry-date").each(function () {
		            var f = $(this);
		            var prevDateSelectionRange = parseInt($("#prevDateSelectionRange").val(), 10); // Convert to integer
		            if (isNaN(prevDateSelectionRange)) { // Fallback if null/undefined/invalid
		                prevDateSelectionRange = 0;
		            }

		            f.datepicker("option", {
		                minDate: -prevDateSelectionRange,  // N days before today
		                maxDate: 0                         // today
		            });

		            if (!f.val()) {
		                f.datepicker("setDate", new Date());
		            }
		        });


		        // EDIT MODE: lock date
		        $(".edit-date").each(function () {
		            var f = $(this);

		            f.prop("readonly", true);

		            f.datepicker("option", "beforeShow", function () {
		                return false;
		            });

		            var t = setInterval(function () {
		                var btn = f.next(".ui-datepicker-trigger");
		                if (btn.length > 0) {
		                    btn.css("pointer-events", "none");
		                    btn.css("opacity", "0.5");
		                    clearInterval(t);
		                }
		            }, 50);
		        });

		        clearInterval(interval);
		    }, 100);

		})();
